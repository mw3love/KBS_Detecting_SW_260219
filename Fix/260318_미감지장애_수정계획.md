           recovery_seconds: float = 0.0, reset_frames: int = 1) -> bool:

# is_abnormal=False, not was_alerting 분기 (현재 L69-73) 대체

    self.just_resolved = False
    self._not_still_count += 1

        self._last_reset_time = now

### ★★★ 2. Heartbeat 로그 확장 + UI 제거

**파일**: `kbs_monitor/ui/main_window.py`
**위치**: `_run_detection()` heartbeat 구간 (L240-244 부근)

**변경**: `self._logger.info()` (파일+UI) → `logging.getLogger()` (파일 전용)
**추가**: 5분마다 각 ROI의 raw 수치를 파일 로그에 덤프

```python
if self._detection_count % 1500 == 0:
    elapsed_min = self._detection_count // 1500 * 5
    _diag_log = logging.getLogger(__name__)
    _diag_log.info("SYSTEM - 감지 정상 실행 중 (%d분 경과)", elapsed_min)
    now = time.time()
    for label, raw in self._detector._last_raw.items():
        still_state = self._detector._still_states.get(label)
        dark_r = raw.get("dark_ratio", -1.0)
        changed_r = raw.get("changed_ratio", -1.0)
        still_timer = still_state.alert_duration if still_state else 0.0
        if still_state and still_state._last_reset_time:
            reset_ago_str = f"직전리셋={now - still_state._last_reset_time:.1f}s전"
        else:
            reset_ago_str = "리셋없음"
        changed_str = (f" changed={changed_r:.1f}%[기준{self._detector.still_changed_ratio}%]"
                       if changed_r >= 0 else "")
        _diag_log.info(
            "DIAG - %s: black=%.1f%%[기준%.0f%%] still_timer=%.1fs[기준%.0fs]%s %s",
            label, dark_r, self._detector.black_dark_ratio,
            still_timer, self._detector.still_duration,
            changed_str, reset_ago_str
        )
```

> 출력 예시:
> `[INFO] SYSTEM - 감지 정상 실행 중 (1440분 경과)`
> `[INFO] DIAG - V2(2TV ON-AIR): black=66.2%[기준95%] still_timer=47.2s[기준60s] changed=2.1%[기준2%] 직전리셋=3.4s전`

### ★★☆ 4. 블랙+모션 억제 (스크롤 자막 오감지 방지)

**파일**: `kbs_monitor/core/detector.py`

**문제**: 스크롤 자막 화면 — 텍스트 면적 2~3% → dark_ratio 97~98% → 95% 기준 초과 → 오감지
**해결**: 움직임(changed_ratio)이 임계값 이상이면 블랙 판정 취소

```python
# Detector.__init__에 추가
self.black_motion_suppress_ratio: float = 0.5  # 0이면 억제 비활성화

# detect_frame() 내 블랙 판정 직후
is_black = dark_ratio >= self.black_dark_ratio
if is_black and self.black_motion_suppress_ratio > 0 and changed_ratio >= 0:
    if changed_ratio >= self.black_motion_suppress_ratio:
        is_black = False
```

**설정 UI**: 감지 설정 탭에 "블랙 모션 억제 비율 (%)" 추가 (0.0~5.0, 기본값 0.5)
**default_config.json**: `"black_motion_suppress_ratio": 0.5` 추가

| 스크롤 자막 | 97% | ~2% | 미감지 ✅ |

### ★★☆ 5. 임계값 근접 경고 (near-miss)

**파일**: `kbs_monitor/core/detector.py`

```python
# Detector.__init__에 추가
self._near_miss_start: Dict[str, float] = {}

# detect_frame() 내 ROI 처리 후
is_near_miss = (dark_ratio > 50.0) or (changed_ratio >= 0 and changed_ratio < 3.0)
if is_near_miss:
    if label not in self._near_miss_start:
        self._near_miss_start[label] = now
    elif now - self._near_miss_start[label] >= 30.0:
        _log.info("NEAR-MISS - ROI[%s]: dark=%.1f%% changed=%.2f%% (30초 지속)",
                  label, dark_ratio, changed_ratio if changed_ratio >= 0 else -1)
        self._near_miss_start[label] = now
else:
    self._near_miss_start.pop(label, None)
```
